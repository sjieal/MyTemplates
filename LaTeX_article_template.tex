		sz = size(aa,2);
		for i=1:sz
		u = mean(aa(:,i),'omitnan');  % 忽略数据中的缺失值计算均值
		sigma = std(aa(:,i),'omitnan');   % 计算标准差  std(x,0,'omitnan')是总体标准差
		lb = u - 3*sigma;    % 区间下界,low bound的缩写
		rb = u + 3*sigma;   % 区间上界,upper bound的缩写
		tmp = (aa(:,i) < lb) | (aa(:,i) > rb);
		ind = find(tmp);
		aa((ind),i)=nan;
		end
		xlswrite('aa.xlsx',aa)
	\end{lstlisting}
