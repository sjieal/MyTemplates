	\caption{Get\underline{~~}Current\underline{~~}Action($Z_t$, $\text{Q}_t$, $epsilon$, strategy)}
	\begin{algorithmic}[1]
		\State initialize $a_t$ = null, $u$\underline{~}planned = False, $r$\underline{~}planned = False, routine = False
		\State $\Omega_{t}$ $\gets$ Get\underline{~~}Action\underline{~~}Space($Z_t$)
		\If{$epsilon > \text{eps}$}
			\State reweight the elements in $\Omega_{t}$ according to the risk preference
			\State randomly select an element from $\Omega_{t}$ as $a_t$
		\Else
			\Repeat
				\State $a_t \gets {\underset{a_t}{\operatorname {arg\,max}}}\,\text{Q}_t$ \hfill//find $a_t$ with the highest Q value in $\text{Q}_t$
				\If{$a_t \in \Omega_{t}$}
					\State break
				\Else
					\State exclude $\text{Q}(Z_t, a_t)$ from $\text{Q}_t$
				\EndIf
			\Until{$\text{Q}_t$ has no element left}
		\EndIf
		\If{$\text{Q}_t = \emptyset$}
			\State \Return $a_t$ \hfill//return null as no action available
		\EndIf
		\If{strategy is \uppercase\expandafter{\romannumeral4}}
			\If{$a_t$ makes $k_{m_j} < 0$ for $\forall{m_j}$} \hfill//module inventory is not enough
				\If{$a_t$ = $u$}
					\State $u$\underline{~}planned $\gets$ True \hfill//action $u$ is planned
				\Else
					\State $r$\underline{~}planned $\gets$ True \hfill//action $r$ is planned
				\EndIf
				\State routine $\gets$ True \hfill//execute routine job to fill in the inventory gap
				\State $\Omega_{t}$ $\gets$ Get\underline{~~}Action\underline{~~}Space($Z_t$) \hfill//re-obtain $\Omega_{t}$ with action $u$ or $r$ planned
				\State randomly select an element from $\Omega_{t}$ as $a_t$
				\State \Return $a_t$
			\EndIf
			\If{$a_t \in \left\{u, r\right\}$}
				\State routine $\gets$ False \hfill//routine job has done
			\EndIf
		\EndIf
		\State \Return $a_t$
	\end{algorithmic}
